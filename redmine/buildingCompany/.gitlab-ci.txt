# .gitlab-ci.yml
stages:
  - build
  - deploy
  - test
  - docs

# Этап 1: Сборка (Build)
build:
  stage: build
  script:
    - echo "===== СБОРКА ПРИЛОЖЕНИЯ ====="
    - pwd
    - dotnet build buildingCompany/buildingCompany/buildingCompany.csproj

# Этап 2: Деплой
# Для dev окружения
deploy_dev:
  stage: deploy
  script:    
    - echo "===== ДЕПЛОЙ НА DEV СРЕДУ ====="
    - |
      $process = Get-Process -Name "buildingCompany" -ErrorAction SilentlyContinue
      if ($process) {
        $process | Stop-Process -Force
        echo "Приложение остановлено"
      } else {
        echo "Приложение не запущено"
      }
    - echo "===== ПУБЛИКАЦИЯ ПРИЛОЖЕНИЯ ====="
    - dotnet publish buildingCompany/buildingCompany/buildingCompany.csproj -c Release -o D:\myapp_deploy_dev\
    - echo "===== ПРИМЕНЕНИЕ МИГРАЦИЙ FLYWAY ДЛЯ DEV ====="
    - D:\tools\flyway\flyway.cmd -url="jdbc:sqlite:D:\myapp_deploy_dev\construction.db" -locations="filesystem:D:/pnrpu/redmine/lab-project/buildingCompany/buildingCompany/MigrationsSQL/dev" -baselineOnMigrate=true migrate
    - echo "===== ЗАПУСК ПРИЛОЖЕНИЯ ====="
    - Start-Process -FilePath "D:\myapp_deploy_dev\buildingCompany.exe" -WorkingDirectory "D:\myapp_deploy_dev" -WindowStyle Hidden
    - echo "===== DEV ДЕПЛОЙ ЗАВЕРШЕН ====="
  only:
    - dev

# Этап 3: Тестирование для DEV
test_dev:
  stage: test
  script:
    - echo "===== ОЖИДАНИЕ ЗАПУСКА ПРИЛОЖЕНИЯ DEV ====="
    - Start-Sleep -Seconds 3
    - echo "===== ЗАПУСК ТЕСТОВ ДЛЯ DEV ====="
    - dotnet test buildingCompany/Test/Test.csproj
    - if ($LASTEXITCODE -ne 0) { exit $LASTEXITCODE }
  dependencies:
    - deploy_dev
  only:
    - dev

# Этап 4: Документация (Docs)
docs:
  stage: docs
  script:
    - echo "===== ОСТАНОВКА СТАРОГО СЕРВЕРА ДОКУМЕНТАЦИИ ====="
    - |
      $processes = Get-Process -Name "docfx" -ErrorAction SilentlyContinue
      if ($processes) {
        $processes | Stop-Process -Force
        echo "Старый сервер документации остановлен"
      } else {
        echo "Сервер документации не запущен"
      }
    - echo "===== ГЕНЕРАЦИЯ ДОКУМЕНТАЦИИ ====="
    - cd buildingCompany
    - docfx docfx.json
    - echo "===== ЗАПУСК СЕРВЕРА ДОКУМЕНТАЦИИ ====="
    - Start-Process -FilePath "docfx" -ArgumentList "serve _site --port 8081" -WindowStyle Hidden
    - echo "===== ДОКУМЕНТАЦИЯ ДОСТУПНА ====="
    - echo "http://localhost:8081"
  only:
    - main
